function readCodexFilterState(config) {
  return config.filters.map(filter => {
    const field =
      document.getElementById(filter.fieldId)?.value ||
      filter.fieldValue;

    const value =
      document.getElementById(filter.id)?.value ||
      filter.selectedValue ||
      "all";

    return {
      field,
      value
    };
  });
}
